# Place this .htaccess in public_html/ (the domain root)
# It rewrites all requests to the swp/ subfolder transparently,
# so simonwickes.com/services/ serves from public_html/swp/services/

RewriteEngine On

# Don't rewrite if the request already targets swp/
RewriteCond %{REQUEST_URI} !^/swp/

# Don't rewrite existing sites
RewriteCond %{REQUEST_URI} !^/resume
RewriteCond %{REQUEST_URI} !^/lamvp

# Don't rewrite system paths
RewriteCond %{REQUEST_URI} !^/\.well-known
RewriteCond %{REQUEST_URI} !^/cgi-bin

# Rewrite everything else into swp/
RewriteRule ^(.*)$ /swp/$1 [L]
